## draft labour exchange and production procedures

Procedure Agent.Production (X : Integer);
var
  I,J : Integer;
begin
  for I := 1 to NumGoods do begin
    X[Produces] := 0
  end;
  for I := 1 to NumGoods do begin
    for J : = NumAgentsF[I] do
      X[Produces] := Power(LabourPurchased,Alpha)
   end;
